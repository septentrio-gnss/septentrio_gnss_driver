52 {
53 
54  // Checking the length first, it should be 19 elements
55  const size_t LENGTH = 19;
56  if (sentence.get_body().size() != LENGTH)
57  {
58  std::stringstream error;
59  error << "Expected GPGSA length is " << LENGTH
60  << ". The actual length is " << sentence.get_body().size();
61  throw ParseException(error.str());
62  }
63 
64  rosaic::GpgsaPtr msg = boost::make_shared<rosaic::Gpgsa>();
65  msg->header.frame_id = frame_id;
66  msg->message_id = sentence.get_body()[0];
67  msg->auto_manual_mode = sentence.get_body()[1];
68  parsing_utilities::ParseUInt8(sentence.get_body()[2], msg->fix_mode);
69  // Words 3-14 of the sentence are SV PRNs. Copying only the non-null strings..
70  msg->sv_ids.resize(12, 0); // 0 is the character needed to fill the new character space, in case 12 (first argument) is larger than sv_ids.
71  size_t n_svs = 0;
72  for (std::vector<std::string>::const_iterator id = sentence.get_body().begin()+3; id < sentence.get_body().begin()+15; ++id)
73  {
74  if (! id->empty())
75  {
76  parsing_utilities::ParseUInt8(*id, msg->sv_ids[n_svs]);
77  ++n_svs;
78  }
79  }
80  msg->sv_ids.resize(n_svs);
81 
82  parsing_utilities::ParseFloat(sentence.get_body()[15], msg->pdop);
83  parsing_utilities::ParseFloat(sentence.get_body()[16], msg->hdop);
84  parsing_utilities::ParseFloat(sentence.get_body()[17], msg->vdop);
85  return msg;
86 }
