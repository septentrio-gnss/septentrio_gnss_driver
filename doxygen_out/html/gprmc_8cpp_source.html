54 rosaic::GprmcPtr GprmcParser::ParseASCII(const NMEASentence& sentence) noexcept(false)
55 {
56 
57  // Checking the length first, it should be between 13 and 14 elements
58  const size_t LEN_MIN = 13;
59  const size_t LEN_MAX = 14;
60 
61  if (sentence.get_body().size() > LEN_MAX ||
62  sentence.get_body().size() < LEN_MIN)
63  {
64  std::stringstream error;
65  error << "Expected GPRMC length is between " << LEN_MIN << " and "
66  << LEN_MAX << ". The actual length is " << sentence.get_body().size();
67  throw ParseException(error.str());
68  }
69 
70  rosaic::GprmcPtr msg = boost::make_shared<rosaic::Gprmc>();
71 
72  msg->header.frame_id = frame_id;
73 
74  msg->message_id = sentence.get_body()[0];
75 
76  if (sentence.get_body()[1].empty() || sentence.get_body()[1] == "0")
77  {
78  msg->utc_seconds = 0;
79  }
80  else
81  {
82  double utc_double;
83  if (string_utilities::ToDouble(sentence.get_body()[1], utc_double))
84  {
85  //ROS_DEBUG("utc_double is %f", (float) utc_double);
86  msg->utc_seconds = parsing_utilities::UTCDoubleToSeconds(utc_double);
87 
88  // The Header's Unix Epoch time stamp
89  time_t unix_time_seconds = parsing_utilities::UTCtoUnix(utc_double);
90  // The following assumes that there are two digits after the decimal point in utc_double, i.e. in the NMEA UTC time.
91  uint32_t unix_time_nanoseconds = (static_cast<uint32_t>(utc_double*100)%100)*10000;
92  msg->header.stamp.sec = unix_time_seconds;
93  msg->header.stamp.nsec = unix_time_nanoseconds;
94  }
95  else
96  {
97  throw ParseException("Error parsing UTC seconds in GPRMC"); // E.g. if one of the fields of the NMEA UTC string is empty
98  }
99  }
100  bool valid = true;
101  bool to_be_ignored = false;
102 
103  msg->position_status = sentence.get_body()[2];
104  // Check to see whether this message should be ignored
105  to_be_ignored &= !(sentence.get_body()[2].compare("A") == 0); // 0 : if both strings are equal.
106  to_be_ignored &= (sentence.get_body()[3].empty() || sentence.get_body()[5].empty());
107 
108  double latitude = 0.0;
109  valid = valid && parsing_utilities::ParseDouble(sentence.get_body()[3], latitude);
110  msg->lat = parsing_utilities::ConvertDMSToDegrees(latitude);
111 
112  double longitude = 0.0;
113  valid = valid && parsing_utilities::ParseDouble(sentence.get_body()[5], longitude);
114  msg->lon = parsing_utilities::ConvertDMSToDegrees(longitude);
115 
116  msg->lat_dir = sentence.get_body()[4];
117  msg->lon_dir = sentence.get_body()[6];
118 
119  valid = valid && parsing_utilities::ParseFloat(sentence.get_body()[7], msg->speed);
120  msg->speed *= KNOTS_TO_MPS;
121 
122  valid = valid && parsing_utilities::ParseFloat(sentence.get_body()[8], msg->track);
123 
124  std::string date_str = sentence.get_body()[9];
125  if (!date_str.empty())
126  {
127  msg->date = std::string("20") + date_str.substr(4, 2) +
128  std::string("-") + date_str.substr(2, 2) +
129  std::string("-") + date_str.substr(0, 2);
130  }
131  valid = valid && parsing_utilities::ParseFloat(sentence.get_body()[10], msg->mag_var);
132  msg->mag_var_direction = sentence.get_body()[11];
133  if (sentence.get_body().size() == LEN_MAX)
134  {
135  msg->mode_indicator = sentence.get_body()[12];
136  }
137 
138  if (!valid)
139  {
140  was_last_gprmc_valid_ = false;
141  throw ParseException("Error parsing GPRMC message.");
142  }
143 
144  was_last_gprmc_valid_ = !to_be_ignored;
145 
146  return msg;
147 }
